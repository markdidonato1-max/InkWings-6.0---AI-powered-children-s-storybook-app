    27тЖТ    27тЖТ// If odd pages, the last image covers just 1 page
    28тЖТ    28тЖТfunction getImagePositions(pageCount: number, imageCount: number): Set<number> {
    29тЖТ    29тЖТ  const positions = new Set<number>()
    30тЖТ    30тЖТ  if (imageCount <= 0 || pageCount <= 0) return positions
    31тЖТ    31тЖТ
    32тЖТ    32тЖТ  // Place images every 2 pages from the start
    33тЖТ    33тЖТ  for (let i = 0; i < imageCount; i++) {
    34тЖТ    34тЖТ    const pos = i * 2
    35тЖТ    35тЖТ    if (pos < pageCount) {
    36тЖТ    36тЖТ      positions.add(pos)
    37тЖТ    37тЖТ    } else {
    38тЖТ    38тЖТ      // If more images than pairs, distribute remaining at the end
    39тЖТ    39тЖТ      const remaining = imageCount - i
    40тЖТ    40тЖТ      const startPos = pageCount - remaining
    41тЖТ    41тЖТ      for (let j = i; j < imageCount; j++) {
    42тЖТ    42тЖТ        positions.add(Math.max(startPos + (j - i), 0))
    43тЖТ    43тЖТ      }
    44тЖТ    44тЖТ      break
    45тЖТ    45тЖТ    }
    46тЖТ    46тЖТ  }
    47тЖТ    47тЖТ  return positions
    48тЖТ    48тЖТ}

    59тЖТ    59тЖТ      combinedText += pages[i].text + ' '
    60тЖТ    60тЖТ    }
    61тЖТ    61тЖТ  }
    62тЖТ    62тЖТ  return combinedText.trim()
    63тЖТ    63тЖТ}
